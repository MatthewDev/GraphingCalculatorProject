044package org.jfree.data.xy;
045
046import org.jfree.data.ComparableObjectItem;
047
048/**
049 * A data item representing data in the form (x, y, deltaX, deltaY), intended
050 * for use by the {@link VectorSeries} class.
051 *
052 * @since 1.0.6
053 */
054public class VectorDataItem extends ComparableObjectItem {
055
056    /**
057     * Creates a new instance of <code>VectorDataItem</code>.
058     *
059     * @param x  the x-value.
060     * @param y  the y-value.
061     * @param deltaX  the vector x.
062     * @param deltaY  the vector y.
063     */
064    public VectorDataItem(double x, double y, double deltaX, double deltaY) {
065        super(new XYCoordinate(x, y), new Vector(deltaX, deltaY));
066    }
067
068    /**
069     * Returns the x-value.
070     *
071     * @return The x-value (never <code>null</code>).
072     */
073    public double getXValue() {
074        XYCoordinate xy = (XYCoordinate) getComparable();
075        return xy.getX();
076    }
077
078    /**
079     * Returns the y-value.
080     *
081     * @return The y-value.
082     */
083    public double getYValue() {
084        XYCoordinate xy = (XYCoordinate) getComparable();
085        return xy.getY();
086    }
087
088    /**
089     * Returns the vector.
090     *
091     * @return The vector (possibly <code>null</code>).
092     */
093    public Vector getVector() {
094        return (Vector) getObject();
095    }
096
097    /**
098     * Returns the x-component for the vector.
099     *
100     * @return The x-component.
101     */
102    public double getVectorX() {
103        Vector vi = (Vector) getObject();
104        if (vi != null) {
105            return vi.getX();
106        }
107        else {
108            return Double.NaN;
109        }
110    }
111
112    /**
113     * Returns the y-component for the vector.
114     *
115     * @return The y-component.
116     */
117    public double getVectorY() {
118        Vector vi = (Vector) getObject();
119        if (vi != null) {
120            return vi.getY();
121        }
122        else {
123            return Double.NaN;
124        }
125    }
126
127}
