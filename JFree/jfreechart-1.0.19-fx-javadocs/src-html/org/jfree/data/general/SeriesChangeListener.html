044
045import java.util.EventListener;
046
047/**
048 * Methods for receiving notification of changes to a data series.
049 */
050public interface SeriesChangeListener extends EventListener {
051
052    /**
053     * Called when an observed series changes in some way.
054     *
055     * @param event  information about the change.
056     */
057    public void seriesChanged(SeriesChangeEvent event);
058
059}
