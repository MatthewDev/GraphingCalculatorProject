050package org.jfree.data;
051
052/**
053 * An interface (optional) that can be implemented by a dataset to assist in
054 * determining the minimum and maximum values.  If not present, 
055 * {@link org.jfree.data.general.DatasetUtilities} will iterate over all the 
056 * values in the dataset to get the bounds.
057 */
058public interface DomainInfo {
059
060    /**
061     * Returns the minimum x-value in the dataset.
062     *
063     * @param includeInterval  a flag that determines whether or not the
064     *                         x-interval is taken into account.
065     *
066     * @return The minimum value or <code>Double.NaN</code> if there are no 
067     *     values.
068     */
069    public double getDomainLowerBound(boolean includeInterval);
070
071    /**
072     * Returns the maximum x-value in the dataset.
073     *
074     * @param includeInterval  a flag that determines whether or not the
075     *                         x-interval is taken into account.
076     *
077     * @return The maximum value or <code>Double.NaN</code> if there are no 
078     *     values.
079     */
080    public double getDomainUpperBound(boolean includeInterval);
081
082    /**
083     * Returns the range of the values in this dataset's domain.
084     *
085     * @param includeInterval  a flag that determines whether or not the
086     *                         x-interval is taken into account.
087     *
088     * @return The range (or <code>null</code> if the dataset contains no
089     *     values).
090     */
091    public Range getDomainBounds(boolean includeInterval);
092
093}
