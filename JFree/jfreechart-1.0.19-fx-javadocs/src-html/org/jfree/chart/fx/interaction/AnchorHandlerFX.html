041package org.jfree.chart.fx.interaction;
042
043import java.awt.geom.Point2D;
044import javafx.scene.input.MouseEvent;
045import org.jfree.chart.fx.ChartCanvas;
046
047/**
048 * Handles mouse clicks on the {@link ChartCanvas} by updating the anchor and 
049 * redrawing the chart.
050 * 
051 * <p>THE API FOR THIS CLASS IS SUBJECT TO CHANGE IN FUTURE RELEASES.  This is
052 * so that we can incorporate feedback on the (new) JavaFX support in 
053 * JFreeChart.</p>
054 * 
055 * @since 1.0.18
056 */
057public class AnchorHandlerFX extends AbstractMouseHandlerFX {
058    
059    /** Records the mouse down location. */
060    private Point2D mousePressedPoint;
061    
062    /**
063     * Creates a new instance.
064     * 
065     * @param id  the id (<code>null</code> not permitted).
066     */
067    public AnchorHandlerFX(String id) {
068        super(id, false, false, false, false);
069    }
070    
071    /**
072     * Handles a mouse pressed event by recording the location of the mouse
073     * pointer (so that later we can check that the click isn't part of a
074     * drag).
075     * 
076     * @param canvas  the chart canvas.
077     * @param e  the mouse event.
078     */
079    @Override
080    public void handleMousePressed(ChartCanvas canvas, MouseEvent e) {
081        this.mousePressedPoint = new Point2D.Double(e.getX(), e.getY());
082    }
083
084    /**
085     * Handles a mouse clicked event by setting the anchor point for the
086     * canvas and redrawing the chart (the anchor point is a reference point
087     * used by the chart to determine crosshair lines).
088     * 
089     * @param canvas  the chart canvas (<code>null</code> not permitted).
090     * @param e  the mouse event (<code>null</code> not permitted).
091     */
092    @Override
093    public void handleMouseClicked(ChartCanvas canvas, MouseEvent e) {
094        if (this.mousePressedPoint == null) {
095            return;
096        }
097        Point2D currPt = new Point2D.Double(e.getX(), e.getY());
098        if (this.mousePressedPoint.distance(currPt) < 2) {
099            canvas.setAnchor(currPt);
100        }
101        this.mousePressedPoint = null;
102    }
103    
104}
