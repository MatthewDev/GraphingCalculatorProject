043package org.jfree.chart.block;
044
045import java.awt.Graphics2D;
046
047import org.jfree.ui.Size2D;
048
049/**
050 * An object that is responsible for arranging a collection of {@link Block}s
051 * within a {@link BlockContainer}.
052 */
053public interface Arrangement {
054
055    /**
056     * Adds a block and a key which can be used to determine the position of
057     * the block in the arrangement.  This method is called by the container
058     * (you don't need to call this method directly) and gives the arrangement
059     * an opportunity to record the details if they are required.
060     *
061     * @param block  the block.
062     * @param key  the key (<code>null</code> permitted).
063     */
064    public void add(Block block, Object key);
065
066    /**
067     * Arranges the blocks within the specified container, subject to the given
068     * constraint.
069     *
070     * @param container  the container (<code>null</code> not permitted).
071     * @param g2  the graphics device.
072     * @param constraint  the constraint.
073     *
074     * @return The container size after the arrangement.
075     */
076    public Size2D arrange(BlockContainer container,
077                          Graphics2D g2,
078                          RectangleConstraint constraint);
079
080    /**
081     * Clears any cached layout information retained by the arrangement.
082     */
083    public void clear();
084
085}
