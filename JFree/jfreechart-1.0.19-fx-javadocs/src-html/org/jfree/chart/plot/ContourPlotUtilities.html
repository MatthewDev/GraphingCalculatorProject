045package org.jfree.chart.plot;
046
047import org.jfree.chart.renderer.xy.XYBlockRenderer;
048import org.jfree.data.Range;
049import org.jfree.data.contour.ContourDataset;
050import org.jfree.data.contour.DefaultContourDataset;
051
052/**
053 * Some utility methods for the {@link ContourPlot} class.
054 *
055 * @deprecated This class is no longer supported (as of version 1.0.4).  If
056 *     you are creating contour plots, please try to use {@link XYPlot} and
057 *     {@link XYBlockRenderer}.
058 */
059public abstract class ContourPlotUtilities {
060
061    /**
062     * Returns the visible z-range.
063     *
064     * @param data  the dataset.
065     * @param x  the x range.
066     * @param y  the y range.
067     *
068     * @return The range.
069     */
070    public static Range visibleRange(ContourDataset data, Range x, Range y) {
071        Range range = null;
072        range = ((DefaultContourDataset) data).getZValueRange(x, y);
073        return range;
074    }
075
076}
