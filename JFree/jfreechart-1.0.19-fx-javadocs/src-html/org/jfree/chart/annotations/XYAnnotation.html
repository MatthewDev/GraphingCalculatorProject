048package org.jfree.chart.annotations;
049
050import java.awt.Graphics2D;
051import java.awt.geom.Rectangle2D;
052
053import org.jfree.chart.axis.ValueAxis;
054import org.jfree.chart.plot.PlotRenderingInfo;
055import org.jfree.chart.plot.XYPlot;
056
057/**
058 * The interface that must be supported by annotations that are to be added to
059 * an {@link XYPlot}.  Note that, in JFreeChart 1.0.14, a non-compatible 
060 * change has been made to this interface (it now extends the Annotation
061 * interface to support change notifications).
062 */
063public interface XYAnnotation extends Annotation {
064
065    /**
066     * Draws the annotation.
067     *
068     * @param g2  the graphics device.
069     * @param plot  the plot.
070     * @param dataArea  the data area.
071     * @param domainAxis  the domain axis.
072     * @param rangeAxis  the range axis.
073     * @param rendererIndex  the renderer index.
074     * @param info  an optional info object that will be populated with
075     *              entity information.
076     */
077    public void draw(Graphics2D g2, XYPlot plot, Rectangle2D dataArea,
078                     ValueAxis domainAxis, ValueAxis rangeAxis,
079                     int rendererIndex, PlotRenderingInfo info);
080
081}
