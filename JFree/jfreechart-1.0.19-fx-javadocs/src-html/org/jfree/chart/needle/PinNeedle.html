046package org.jfree.chart.needle;
047
048import java.awt.Graphics2D;
049import java.awt.geom.Area;
050import java.awt.geom.Ellipse2D;
051import java.awt.geom.GeneralPath;
052import java.awt.geom.Point2D;
053import java.awt.geom.Rectangle2D;
054import java.io.Serializable;
055
056/**
057 * A needle that is drawn as a pin shape.
058 */
059public class PinNeedle extends MeterNeedle
060                       implements Cloneable, Serializable {
061
062    /** For serialization. */
063    private static final long serialVersionUID = -3787089953079863373L;
064
065    /**
066     * Draws the needle.
067     *
068     * @param g2  the graphics device.
069     * @param plotArea  the plot area.
070     * @param rotate  the rotation point.
071     * @param angle  the angle.
072     */
073    @Override
074    protected void drawNeedle(Graphics2D g2, Rectangle2D plotArea,
075                              Point2D rotate, double angle) {
076
077        Area shape;
078        GeneralPath pointer = new GeneralPath();
079
080        int minY = (int) (plotArea.getMinY());
081        //int maxX = (int) (plotArea.getMaxX());
082        int maxY = (int) (plotArea.getMaxY());
083        int midX = (int) (plotArea.getMinX() + (plotArea.getWidth() / 2));
084        //int midY = (int) (plotArea.getMinY() + (plotArea.getHeight() / 2));
085        int lenX = (int) (plotArea.getWidth() / 10);
086        if (lenX < 2) {
087            lenX = 2;
088        }
089
090        pointer.moveTo(midX - lenX, maxY - lenX);
091        pointer.lineTo(midX + lenX, maxY - lenX);
092        pointer.lineTo(midX, minY + lenX);
093        pointer.closePath();
094
095        lenX = 4 * lenX;
096        Ellipse2D circle = new Ellipse2D.Double(midX - lenX / 2,
097                plotArea.getMaxY() - lenX, lenX, lenX);
098
099        shape = new Area(circle);
100        shape.add(new Area(pointer));
101        if ((rotate != null) && (angle != 0)) {
102            /// we have rotation
103            getTransform().setToRotation(angle, rotate.getX(), rotate.getY());
104            shape.transform(getTransform());
105        }
106
107        defaultDisplay(g2, shape);
108
109    }
110
111    /**
112     * Tests another object for equality with this object.
113     *
114     * @param obj  the object to test (<code>null</code> permitted).
115     *
116     * @return A boolean.
117     */
118    @Override
119    public boolean equals(Object obj) {
120        if (obj == this) {
121            return true;
122        }
123        if (!(obj instanceof PinNeedle)) {
124            return false;
125        }
126        if (!super.equals(obj)) {
127            return false;
128        }
129        return true;
130    }
131
132    /**
133     * Returns a hash code for this instance.
134     *
135     * @return A hash code.
136     */
137    @Override
138    public int hashCode() {
139        return super.hashCode();
140    }
141
142    /**
143     * Returns a clone of this needle.
144     *
145     * @return A clone.
146     *
147     * @throws CloneNotSupportedException if the <code>PinNeedle</code>
148     *     cannot be cloned (in theory, this should not happen).
149     */
150    @Override
151    public Object clone() throws CloneNotSupportedException {
152        return super.clone();
153    }
154
155}
