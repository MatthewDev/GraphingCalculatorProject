046package org.jfree.chart.needle;
047
048import java.awt.Graphics2D;
049import java.awt.geom.GeneralPath;
050import java.awt.geom.Point2D;
051import java.awt.geom.Rectangle2D;
052import java.io.Serializable;
053
054/**
055 * A needle in the shape of a pointer, for use with the
056 * {@link org.jfree.chart.plot.CompassPlot} class.
057 */
058public class PointerNeedle extends MeterNeedle
059                           implements Cloneable, Serializable {
060
061    /** For serialization. */
062    private static final long serialVersionUID = -4744677345334729606L;
063
064    /**
065     * Draws the needle.
066     *
067     * @param g2  the graphics device.
068     * @param plotArea  the plot area.
069     * @param rotate  the rotation point.
070     * @param angle  the angle.
071     */
072    @Override
073    protected void drawNeedle(Graphics2D g2, Rectangle2D plotArea,
074                              Point2D rotate, double angle) {
075
076        GeneralPath shape1 = new GeneralPath();
077        GeneralPath shape2 = new GeneralPath();
078        float minX = (float) plotArea.getMinX();
079        float minY = (float) plotArea.getMinY();
080        float maxX = (float) plotArea.getMaxX();
081        float maxY = (float) plotArea.getMaxY();
082        float midX = (float) (minX + (plotArea.getWidth() / 2));
083        float midY = (float) (minY + (plotArea.getHeight() / 2));
084
085        shape1.moveTo(minX, midY);
086        shape1.lineTo(midX, minY);
087        shape1.lineTo(maxX, midY);
088        shape1.closePath();
089
090        shape2.moveTo(minX, midY);
091        shape2.lineTo(midX, maxY);
092        shape2.lineTo(maxX, midY);
093        shape2.closePath();
094
095        if ((rotate != null) && (angle != 0)) {
096            /// we have rotation huston, please spin me
097            getTransform().setToRotation(angle, rotate.getX(), rotate.getY());
098            shape1.transform(getTransform());
099            shape2.transform(getTransform());
100        }
101
102        if (getFillPaint() != null) {
103            g2.setPaint(getFillPaint());
104            g2.fill(shape1);
105        }
106
107        if (getHighlightPaint() != null) {
108            g2.setPaint(getHighlightPaint());
109            g2.fill(shape2);
110        }
111
112        if (getOutlinePaint() != null) {
113            g2.setStroke(getOutlineStroke());
114            g2.setPaint(getOutlinePaint());
115            g2.draw(shape1);
116            g2.draw(shape2);
117        }
118    }
119
120    /**
121     * Tests another object for equality with this object.
122     *
123     * @param obj  the object to test (<code>null</code> permitted).
124     *
125     * @return A boolean.
126     */
127    @Override
128    public boolean equals(Object obj) {
129        if (obj == this) {
130            return true;
131        }
132        if (!(obj instanceof PointerNeedle)) {
133            return false;
134        }
135        if (!super.equals(obj)) {
136            return false;
137        }
138        return true;
139    }
140
141    /**
142     * Returns a hash code for this instance.
143     *
144     * @return A hash code.
145     */
146    @Override
147    public int hashCode() {
148        return super.hashCode();
149    }
150
151    /**
152     * Returns a clone of this needle.
153     *
154     * @return A clone.
155     *
156     * @throws CloneNotSupportedException if the <code>PointerNeedle</code>
157     *     cannot be cloned (in theory, this should not happen).
158     */
159    @Override
160    public Object clone() throws CloneNotSupportedException {
161        return super.clone();
162    }
163
164}
