041package org.jfree.chart.axis;
042
043/**
044 * An interface used by the {@link DateAxis} and {@link NumberAxis} classes to
045 * obtain a suitable {@link TickUnit}.
046 */
047public interface TickUnitSource {
048
049    /**
050     * Returns the smallest tick unit available in the source that is larger
051     * than <code>unit</code> or, if there is no larger unit, returns 
052     * <code>unit</code>.
053     *
054     * @param unit  the unit (<code>null</code> not permitted).
055     *
056     * @return A tick unit that is larger than the supplied unit.
057     */
058    public TickUnit getLargerTickUnit(TickUnit unit);
059
060    /**
061     * Returns the tick unit in the collection that is greater than or equal
062     * to (in size) the specified unit.
063     *
064     * @param unit  the unit.
065     *
066     * @return A unit from the collection.
067     */
068    public TickUnit getCeilingTickUnit(TickUnit unit);
069
070    /**
071     * Returns the smallest tick unit available in the source that is greater 
072     * than or equal to the specified size.  If there is no such tick unit,
073     * the method should return the largest available tick in the source.
074     *
075     * @param size  the size.
076     *
077     * @return A unit from the collection (never <code>null</code>).
078     */
079    public TickUnit getCeilingTickUnit(double size);
080
081}
