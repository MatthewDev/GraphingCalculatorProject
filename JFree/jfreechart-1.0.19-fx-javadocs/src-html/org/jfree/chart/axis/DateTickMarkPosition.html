041package org.jfree.chart.axis;
042
043import java.io.ObjectStreamException;
044import java.io.Serializable;
045
046/**
047 * Used to indicate the required position of tick marks on a date axis relative
048 * to the underlying time period.
049 */
050public final class DateTickMarkPosition implements Serializable {
051
052    /** For serialization. */
053    private static final long serialVersionUID = 2540750672764537240L;
054
055    /** Start of period. */
056    public static final DateTickMarkPosition START
057        = new DateTickMarkPosition("DateTickMarkPosition.START");
058
059    /** Middle of period. */
060    public static final DateTickMarkPosition MIDDLE
061        = new DateTickMarkPosition("DateTickMarkPosition.MIDDLE");
062
063    /** End of period. */
064    public static final DateTickMarkPosition END
065        = new DateTickMarkPosition("DateTickMarkPosition.END");
066
067    /** The name. */
068    private String name;
069
070    /**
071     * Private constructor.
072     *
073     * @param name  the name.
074     */
075    private DateTickMarkPosition(String name) {
076        this.name = name;
077    }
078
079    /**
080     * Returns a string representing the object.
081     *
082     * @return The string.
083     */
084    @Override
085    public String toString() {
086        return this.name;
087    }
088
089    /**
090     * Returns <code>true</code> if this object is equal to the specified
091     * object, and <code>false</code> otherwise.
092     *
093     * @param obj  the other object.
094     *
095     * @return A boolean.
096     */
097    @Override
098    public boolean equals(Object obj) {
099
100        if (this == obj) {
101            return true;
102        }
103        if (!(obj instanceof DateTickMarkPosition)) {
104            return false;
105        }
106        DateTickMarkPosition position = (DateTickMarkPosition) obj;
107        if (!this.name.equals(position.toString())) {
108            return false;
109        }
110        return true;
111
112    }
113
114    /**
115     * Ensures that serialization returns the unique instances.
116     *
117     * @return The object.
118     *
119     * @throws ObjectStreamException if there is a problem.
120     */
121    private Object readResolve() throws ObjectStreamException {
122        if (this.equals(DateTickMarkPosition.START)) {
123            return DateTickMarkPosition.START;
124        }
125        else if (this.equals(DateTickMarkPosition.MIDDLE)) {
126            return DateTickMarkPosition.MIDDLE;
127        }
128        else if (this.equals(DateTickMarkPosition.END)) {
129            return DateTickMarkPosition.END;
130        }
131        return null;
132    }
133
134
135}
