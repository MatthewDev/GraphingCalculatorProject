041package org.jfree.chart.labels;
042
043import org.jfree.data.general.PieDataset;
044
045/**
046 * A tool tip generator that is used by the
047 * {@link org.jfree.chart.plot.PiePlot} class.
048 */
049public interface PieToolTipGenerator {
050
051    /**
052     * Generates a tool tip text item for the specified item in the dataset.
053     * This method can return <code>null</code> to indicate that no tool tip
054     * should be displayed for an item.
055     *
056     * @param dataset  the dataset (<code>null</code> not permitted).
057     * @param key  the section key (<code>null</code> not permitted).
058     *
059     * @return The tool tip text (possibly <code>null</code>).
060     */
061    public String generateToolTip(PieDataset dataset, Comparable key);
062
063}
