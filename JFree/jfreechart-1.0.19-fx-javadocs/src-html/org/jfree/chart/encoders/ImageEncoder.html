042package org.jfree.chart.encoders;
043
044import java.awt.image.BufferedImage;
045import java.io.IOException;
046import java.io.OutputStream;
047
048/**
049 * Interface for abstracting different types of image encoders.
050 */
051public interface ImageEncoder {
052
053    /**
054     * Encodes an image in a particular format.
055     *
056     * @param bufferedImage  The image to be encoded.
057     *
058     * @return The byte[] that is the encoded image.
059     *
060     * @throws IOException if there is an IO problem.
061     */
062    public byte[] encode(BufferedImage bufferedImage) throws IOException;
063
064    /**
065     * Encodes an image in a particular format and writes it to an OutputStream.
066     *
067     * @param bufferedImage  The image to be encoded.
068     * @param outputStream  The OutputStream to write the encoded image to.
069     * @throws IOException if there is an IO problem.
070     */
071    public void encode(BufferedImage bufferedImage, OutputStream outputStream)
072        throws IOException;
073
074    /**
075     * Get the quality of the image encoding.
076     *
077     * @return A float representing the quality.
078     */
079    public float getQuality();
080
081    /**
082     * Set the quality of the image encoding (not supported by all
083     * ImageEncoders).
084     *
085     * @param quality  A float representing the quality.
086     */
087    public void setQuality(float quality);
088
089    /**
090     * Get whether the encoder should encode alpha transparency.
091     *
092     * @return Whether the encoder is encoding alpha transparency.
093     */
094    public boolean isEncodingAlpha();
095
096    /**
097     * Set whether the encoder should encode alpha transparency (not
098     * supported by all ImageEncoders).
099     *
100     * @param encodingAlpha  Whether the encoder should encode alpha
101     *                       transparency.
102     */
103    public void setEncodingAlpha(boolean encodingAlpha);
104
105}
